###############################################################
# Install documentation
# Build API docs, if doxygen is available
###############################################################

# Set docs target directory
if (WIN32)
  set(GU_DOCS_PATH "Docs")
elseif (UNIX)
  set(GU_DOCS_PATH "share/GU/docs")
elseif (APPLE)
  # TODO
endif ()
  
if (GU_INSTALL_DOCS)
  install(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/
    DESTINATION ${GU_DOCS_PATH}
    PATTERN ".svn" EXCLUDE
    PATTERN "_svn" EXCLUDE
    PATTERN "src" EXCLUDE
  )
endif ()

# Build and install API documentation if doxygen is available
if(DOXYGEN_FOUND)
	configure_file(${GU_TEMPLATES_DIR}/html.cfg.in ${GU_BINARY_DIR}/html.cfg)
	add_custom_target(doc COMMAND 
    ${DOXYGEN_EXECUTABLE} ${GU_BINARY_DIR}/html.cfg 
    WORKING_DIRECTORY ${GU_SOURCE_DIR}/Docs/src/ 
    COMMENT "Building documentation" VERBATIM
  )
	file(WRITE ${GU_BINARY_DIR}/api/temp.txt "This file should be deleted before installation. It is only here to force the creation of the api folder")
	file(REMOVE ${GU_BINARY_DIR}/api/temp.txt)
	install(DIRECTORY ${GU_BINARY_DIR}/api DESTINATION ${GU_DOCS_PATH})
endif()
